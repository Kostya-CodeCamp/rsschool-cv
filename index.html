                                    "one": "hello",
                                    "two": "mahai",
                                    "five": "hi",
                                    "test": 21,
                                    "odd": "hi",
                                    "mix": "mix"
                                };


                                function f4() {
                                    let out ='';
                                    for(key in a4){

                                    }
